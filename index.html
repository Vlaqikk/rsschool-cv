Heap Sort is a comparison-based sorting algorithm that transforms an array into a heap data structure. A heap is a complete binary tree where each node satisfies the heap property, meaning that the parent node is either greater than or equal to (max heap) or less than or equal to (min heap) its child nodes.

The algorithm consists of two main parts:

Heapify: This function ensures that the subtree rooted at node i maintains the heap property. If the heap property is violated at i, it corrects it by swapping i with its largest (for max heap) or smallest (for min heap) child recursively.

Heap Sort Utilization: It first builds a max heap (for ascending order) from the input array. After building the heap, the root element (the largest) is swapped with the last element in the heap and then heapified to adjust the heap structure. This process is repeated for the remaining elements, resulting in a sorted array.

Heap Sort has a time complexity of O(n log n) for all cases, making it efficient and optimal for sorting large datasets.

    function heapSort(arr) {
        function heapify(arr, n, i) {
            let largest = i;
            let left = 2 * i + 1;
            let right = 2 * i + 2;

            if (left < n && arr[left] > arr[largest]) {
                largest = left;
            }

            if (right < n && arr[right] > arr[largest]) {
                largest = right;
            }

            if (largest !== i) {
                [arr[i], arr[largest]] = [arr[largest], arr[i]];
                heapify(arr, n, largest);
            }
        }

        function heapSortUtil(arr) {
            const n = arr.length;

            for (let i = Math.floor(n / 2) - 1; i >= 0; i--) {
                heapify(arr, n, i);
            }

            for (let i = n - 1; i > 0; i--) {
                [arr[0], arr[i]] = [arr[i], arr[0]];
                heapify(arr, i, 0);
            }
        }

        heapSortUtil(arr);
    }

    let arr = [12, 11, 13, 5, 6, 7];
    console.log("Source array:", arr);
    heapSort(arr);
    console.log("Sorted array:", arr);
